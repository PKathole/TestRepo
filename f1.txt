import java.util.Scanner;

public class Percentage {
	public static void main(String[] args) {
		float n,S1,S2,S3,S4,S5,Sum,Percentage;
		System.out.println("Enter the marks of 5 subject : ");
		Scanner sc=new Scanner(System.in);
		System.out.println("Enter the marks of S1 subject : ");
		S1=sc.nextInt();
		System.out.println("Enter the marks of S2 subject : ");
		S2=sc.nextInt();
		System.out.println("Enter the marks of S3 subject : ");
		S3=sc.nextInt();
		System.out.println("Enter the marks of S4 subject : ");
